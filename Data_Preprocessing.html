def create_mkdir(dirname):
    if os.path.exists(dirname):
        shutil.rmtree(dirname)
    os.mkdir(dirname)

def file_move(data,old_dir,new_dir):
    for filename in data:
        shutil.move(old_dir+filename, new_dir+filename)
